def minimum_spanning_tree(adjacency_matrix):
    n = len(adjacency_matrix)
    selected = [False] * n  # –í–µ—Ä—à–∏–Ω—ã, –≤–∫–ª—é—á–µ–Ω–Ω—ã–µ –≤ –æ—Å—Ç–æ–≤–Ω–æ–µ –¥–µ—Ä–µ–≤–æ

    num_selected = 1
    edges = []

    while num_selected < n:
        min_weight = float('inf')
        min_edge = None

            edges.append(min_edge)

    return edges

–°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ —Å–ª–∏—è–Ω–∏–µ–º (Merge Sort) –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ –∏–º–µ–µ—Ç –≤—Ä–µ–º–µ–Ω–Ω—É—é —Å–ª–æ–∂–Ω–æ—Å—Ç—å –≤–∏–¥–∞ O(n log n), —á—Ç–æ –¥–µ–ª–∞–µ—Ç –µ–µ –±–æ–ª–µ–µ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–π –ø–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—é —Å –∫–≤–∞–¥—Ä–∞—Ç–∏—á–Ω—ã–º–∏ –∞–ª–≥–æ—Ä–∏—Ç–º–∞–º–∏ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏, —Ç–∞–∫–∏–º–∏ –∫–∞–∫ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –≤—ã–±–æ—Ä–æ–º (Selection Sort) —Å –≤—Ä–µ–º–µ–Ω–Ω–æ–π —Å–ª–æ–∂–Ω–æ—Å—Ç—å—é O(n^2).

–î–ª—è —Å—Ä–∞–≤–Ω–µ–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏ –∞–ª–≥–æ—Ä–∏—Ç–º–æ–≤, –¥–∞–≤–∞–π—Ç–µ –Ω–∞–ø–∏—à–µ–º –ø—Å–µ–≤–¥–æ–∫–æ–¥ –¥–ª—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏ —Å–ª–∏—è–Ω–∏–µ–º –∏ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏ –≤—ã–±–æ—Ä–æ–º, –∞ –∑–∞—Ç–µ–º —Å–≥–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Ç–µ—Å—Ç–æ–≤—ã–µ –ø—Ä–∏–º–µ—Ä—ã –¥–ª—è —Å—Ä–∞–≤–Ω–µ–Ω–∏—è.

```java
merge_sort(arr):
    if length(arr) <= 1:
        return arr

    middle = length(arr) / 2
    left = arr[0...middle-1]
    right = arr[middle...]

    left = merge_sort(left)
    right = merge_sort(right)

    return merge(left, right)

merge(left, right):
    result = []
    while left is not empty and right is not empty:
        if left[0] <= right[0]:
            result.append(left[0])
            left = left[1...]
        else:
            result.append(right[0])
            right = right[1...]

    if left is not empty:
        result.extend(left)
    if right is not empty:
        result.extend(right)

    return result

```

selection_sort(arr):
    for i from 0 to length(arr)-1:
        min_index = i
        for j from i+1 to length(arr):
            if arr[j] < arr[min_index]:
                min_index = j
        swap(arr[i], arr[min_index])

#### –†–∞–∑–¥–µ–ª–∏—Ç—å –∏ —É–ø–∞–∫–æ–≤–∞—Ç—å

–ó–∞–¥–∞—á—É –æ —Ä—é–∫–∑–∞–∫–µ (—Å–º. —Ä–∞–∑–¥–µ–ª ¬´–ü–æ–ª–Ω—ã–π –ø–µ—Ä–µ–±–æ—Ä¬ª ) —Ç–æ–∂–µ –º–æ–∂–Ω–æ —Ä–∞–∑–¥–µ–ª–∏—Ç—å –∏ —Ç–µ–º —Å–∞–º—ã–º —Ä–µ—à–∏—Ç—å. –ï—Å–ª–∏ –≤—ã –Ω–µ –∑–∞–±—ã–ª–∏, —É –Ω–∞—Å $n$ –ø—Ä–µ–¥–º–µ—Ç–æ–≤ –Ω–∞ –≤—ã–±–æ—Ä. –ú—ã –æ–±–æ–∑–Ω–∞—á–∏–º —Å–≤–æ–π—Å—Ç–≤–æ –∫–∞–∂–¥–æ–≥–æ –∏–∑ –Ω–∏—Ö —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:

–ò–Ω–¥–µ–∫—Å $i$ –ø—Ä–µ–¥–º–µ—Ç–∞ –º–æ–∂–µ—Ç –±—ã—Ç—å –ª—é–±—ã–º —á–∏—Å–ª–æ–º –æ—Ç $1$ –¥–æ $n$. –ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –¥–æ—Ö–æ–¥ –¥–ª—è –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç–∏ c —Ä—é–∫–∑–∞–∫–∞ —Å —É–∂–µ –≤—ã–±—Ä–∞–Ω–Ω—ã–º–∏ n –ø—Ä–µ–¥–º–µ—Ç–∞–º–∏ —Å–æ—Å—Ç–∞–≤–ª—è–µ—Ç $K(n,¬†c)$. –ï—Å–ª–∏ —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ—Ç—Å—è –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–π –ø—Ä–µ–¥–º–µ—Ç $i = n¬†+ 1$, —Ç–æ –æ–Ω –ª–∏–±–æ –ø–æ–≤—ã—Å–∏—Ç, –ª–∏–±–æ –Ω–µ –ø–æ–≤—ã—Å–∏—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ –≤–æ–∑–º–æ–∂–Ω—ã–π –¥–æ—Ö–æ–¥, –∫–æ—Ç–æ—Ä—ã–π —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è —Ä–∞–≤–Ω—ã–º –±–æ–ª—å—à–µ–º—É –∏–∑ –¥–≤—É—Ö –∑–Ω–∞—á–µ–Ω–∏–π.

2. $K(n, c ‚àí w_{n+1}) + v_{n+1}$ ‚Äî –µ—Å–ª–∏ –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–π –ø—Ä–µ–¥–º–µ—Ç –≤—ã–±—Ä–∞–Ω.

–≠—Ç–æ –∑–Ω–∞—á–∏—Ç, —á—Ç–æ –º—ã –º–æ–∂–µ–º –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å —Ä–µ—à–µ–Ω–∏–µ –¥–ª—è n –ø—Ä–µ–¥–º–µ—Ç–æ–≤ –∫–∞–∫ –º–∞–∫—Å–∏–º—É–º —á–∞—Å—Ç–Ω—ã—Ö —Ä–µ—à–µ–Ω–∏–π –¥–ª—è$ n¬†‚Äì 1$ –ø—Ä–µ–¥–º–µ—Ç–æ–≤: 
$$K(n, c) = max (K(n ‚àí 1, c), K(n ‚àí 1, c ‚àí w_n) + v_n)$$

```python
def knapsack_memoization(values, weights, capacity):
    n = len(values)
    # –°–æ–∑–¥–∞–µ–º –¥–≤—É–º–µ—Ä–Ω—ã–π –º–∞—Å—Å–∏–≤ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤.
    # dp[i][w] –±—É–¥–µ—Ç —Ö—Ä–∞–Ω–∏—Ç—å –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø—Ä–µ–¥–º–µ—Ç–æ–≤,
    # –∫–æ—Ç–æ—Ä—ã–µ –º–æ–∂–Ω–æ —É–ø–∞–∫–æ–≤–∞—Ç—å –≤ —Ä—é–∫–∑–∞–∫ —Å –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç—å—é w, –∏—Å–ø–æ–ª—å–∑—É—è –ø–µ—Ä–≤—ã–µ i –ø—Ä–µ–¥–º–µ—Ç–æ–≤.
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]

    # –ó–∞–ø–æ–ª–Ω—è–µ–º dp —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –º–µ–º–æ–∏–∑–∞—Ü–∏–∏ (–¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è)
    for i in range(1, n + 1):
        for w in range(1, capacity + 1):
            if weights[i - 1] <= w:
                # –ï—Å–ª–∏ –≤–µ—Å –ø—Ä–µ–¥–º–µ—Ç–∞ –º–µ–Ω—å—à–µ –∏–ª–∏ —Ä–∞–≤–µ–Ω —Ç–µ–∫—É—â–µ–π –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç–∏, –º—ã –º–æ–∂–µ–º —Ä–∞—Å—Å–º–æ—Ç—Ä–µ—Ç—å –¥–≤–∞ –≤–∞—Ä–∏–∞–Ω—Ç–∞:
                # 1. –í–∫–ª—é—á–∏—Ç—å –ø—Ä–µ–¥–º–µ—Ç –≤ —Ä—é–∫–∑–∞–∫, —Ç–æ–≥–¥–∞ —Å—Ç–æ–∏–º–æ—Å—Ç—å –±—É–¥–µ—Ç —Ä–∞–≤–Ω–∞ values[i-1] + dp[i-1][w-weights[i-1]].
                # 2. –ù–µ –≤–∫–ª—é—á–∞—Ç—å –ø—Ä–µ–¥–º–µ—Ç –≤ —Ä—é–∫–∑–∞–∫, —Ç–æ–≥–¥–∞ —Å—Ç–æ–∏–º–æ—Å—Ç—å –æ—Å—Ç–∞–µ—Ç—Å—è —Ä–∞–≤–Ω–æ–π dp[i-1][w].
                dp[i][w] = max(values[i - 1] + dp[i - 1][w - weights[i - 1]], dp[i - 1][w])
            else:
                # –ï—Å–ª–∏ –≤–µ—Å –ø—Ä–µ–¥–º–µ—Ç–∞ –±–æ–ª—å—à–µ, —á–µ–º —Ç–µ–∫—É—â–∞—è –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç—å, —Ç–æ –º—ã –Ω–µ –º–æ–∂–µ–º –≤–∫–ª—é—á–∏—Ç—å –µ–≥–æ –≤ —Ä—é–∫–∑–∞–∫.
                dp[i][w] = dp[i - 1][w]

    # –í–æ—Å—Å—Ç–∞–Ω–æ–≤–∏–º –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ, –æ–ø—Ä–µ–¥–µ–ª—è—è, –∫–∞–∫–∏–µ –ø—Ä–µ–¥–º–µ—Ç—ã –≤–∫–ª—é—á–µ–Ω—ã –≤ —Ä—é–∫–∑–∞–∫.
    selected_items = []
    i, w = n, capacity
    while i > 0 and w > 0:
        if dp[i][w] != dp[i - 1][w]:
            selected_items.append(i - 1)
            w -= weights[i - 1]
        i -= 1

    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é —Å—Ç–æ–∏–º–æ—Å—Ç—å –∏ —Å–ø–∏—Å–æ–∫ –≤—ã–±—Ä–∞–Ω–Ω—ã—Ö –ø—Ä–µ–¥–º–µ—Ç–æ–≤
    return dp[n][capacity], selected_items

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
values = [60, 100, 120]
weights = [10, 20, 30]
capacity = 50

max_value, selected_items = knapsack_memoization(values, weights, capacity)
print("–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å:", max_value)
print("–í—ã–±—Ä–∞–Ω–Ω—ã–µ –ø—Ä–µ–¥–º–µ—Ç—ã:", selected_items)

```
